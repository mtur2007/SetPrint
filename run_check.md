        ((),[])]

# sq > mp [ シーケンス型の値が最大長になるか ]

        ({},[])]

# mp > sq [ マッピング型の値が最大長になるか ]

        ((),{})]

# < mp : mp >

# マッピング型同士での可変表示調査

        ({},{})]

#=======================================================
